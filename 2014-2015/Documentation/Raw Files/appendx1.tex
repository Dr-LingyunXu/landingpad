\section{Walkthroughs}\label{walkthroughs}

\noindent Tool Installation Walkthroughs

\noindent These instructions assume you are using Ubuntu 14.04.

%----------------------------------------------------------------------------------------------------------------------------------------------------------------------------

\subsection{Installing ROS Indigo}

\noindent ROS Wiki Page:
\begin{itemize}
	\item[]\href{http://wiki.ros.org/indigo/Installation/Ubuntu}{$http://wiki.ros.org/indigo/Installation/Ubuntu$}
\end{itemize}

\noindent 1. Setup your sources.list
\begin{itemize}
	\item[]sudo sh -c 'echo "deb http://packages.ros.org/ros/ubuntu trusty main"
	\item[] $>$ /etc/apt/sources.list.d/ros-latest.list'
\end{itemize}

\noindent 2. Set up your keys using:
\begin{itemize}
	\item[]wget https://raw.githubusercontent.com/ros/rosdistro/master/ros.key
	\item[] -O - | sudo apt-key add -
\end{itemize}

\noindent 3. Update your system using:
\begin{itemize}
	\item[]sudo apt-get update
\end{itemize}

\noindent 4. Install ROS Indigo using:
\begin{itemize}
	\item[]sudo apt-get install ros-indigo-desktop-full
\end{itemize}

\noindent 5. Initialize Rosdep using:
\begin{itemize}
	\item[]sudo rosdep init
	\item[]rosdep update
\end{itemize}

\noindent 6. Setup your environment using:
\begin{itemize}
	\item[]echo "source /opt/ros/indigo/setup.bash" $>>$ \textasciitilde /.bashrc
	\item[]source \textasciitilde /.bashrc
\end{itemize}

\noindent 7. Get ROS install using:
\begin{itemize}
	\item[]sudo apt-get install python-rosinstall
\end{itemize}

%----------------------------------------------------------------------------------------------------------------------------------------------------------------------------

\subsection{Creating a Catkin Workspace}

\noindent Catkin Workspace Wiki Page:
\begin{itemize}
	\item[]\href{http://wiki.ros.org/catkin/Tutorials/create_a_workspace}{$http://wiki.ros.org/catkin/Tutorials/create\_a\_workspace$}
\end{itemize}

\noindent 1. Source the setup.bash file in your ROS directory by typing the following command in a terminal window:
\begin{itemize}
	\item[]source /opt/ros/indigo/setup.bash
\end{itemize}

\noindent 2. Make the Catkin Workspace directory and the source subdirectory using:
\begin{itemize}
	\item[]mkdir -p catkin\textunderscore ws/src
\end{itemize}

\noindent 3. Change directories into catkin\textunderscore  ws using:
\begin{itemize}
	\item[]cd catkin\textunderscore ws/src
\end{itemize}

\noindent 4. Initialize your catkin workspace using:
\begin{itemize}
	\item[]catkin\textunderscore init\textunderscore workspace
\end{itemize}

\noindent 5. Change directories back into your catkin workspace directory using:
\begin{itemize}
	\item[]cd ..
\end{itemize}

\noindent 6. Run catkin\textunderscore make command using:
\begin{itemize}
	\item[]catkin\textunderscore make
\end{itemize}

\noindent 7. Source the new setup.bash file that was created within your catkin workspace using:
\begin{itemize}
	\item[]source devel/setup.bash
\end{itemize}

\subsection{Installing usb\textunderscore cam node}

\noindent 1. Clone github repository for usb\textunderscore cam node into /catkin\textunderscore ws/src using:
\begin{itemize}
	\item[]git clone https://github.com/bosch-ros-pkg/usb\textunderscore cam.git
\end{itemize}

\noindent 2. Build usb\textunderscore cam node using the following command in the /catkin\textunderscore ws/ directory:
\begin{itemize}
	\item[]catkin\textunderscore make
\end{itemize}

\subsection{Installing PTAM}

\noindent PTAM Wiki Page:
\begin{itemize}
	\item[]\href{http://wiki.ros.org/ethzasl_ptam}{$http://wiki.ros.org/ethzasl\_ptam$}
\end{itemize}

\noindent 1. Change directories into catkin\textunderscore ws/src using:
\begin{itemize}
	\item[]cd ~/catkin\textunderscore ws/src
\end{itemize}

\noindent 2. Clone the ethzasl\textunderscore ptam repository from github using:
\begin{itemize}
	\item[]git clone git://github.com/ethz-asl/ethzasl\textunderscore ptam.git ethzasl\textunderscore ptam
\end{itemize}

\noindent 3. Get necessary libraries for PTAM using:
\begin{itemize}
	\item[]sudo apt-get install freeglut3 freeglut3-dev libblas3 libblas-dev liblapack3 liblapack-dev
\end{itemize}

\noindent 4. Build PTAM using the following command in your catkin\textunderscore ws directory:
\begin{itemize}
	\item[]catkin\textunderscore make
\end{itemize}

\subsection{Installing OpenCV}

\noindent OpenCV Online Reference Manual:
\begin{itemize}
	\item[]\href{http://docs.opencv.org/}{$http://docs.opencv.org/$}
\end{itemize}

\noindent 1. Ensure you have the essential build tools needed for OpenCV:
\begin{itemize}
	\item[]sudo apt-get install build-essential cmake pkg-config
\end{itemize}

\noindent 2. Install required Image I/O Libraries:
\begin{itemize}
	\item[]sudo apt-get install libjpeg62-dev libtiff4-dev libjasper-dev
\end{itemize}

\noindent 3. Install gtk highgui Libraries:
\begin{itemize}
	\item[]sudo apt-get install  libgtk2.0-dev
\end{itemize}

\noindent 4. Install video Libraries:
\begin{itemize}
	\item[]sudo apt-get install libavcodec-dev libavformat-dev libswscale-dev libv4l-dev
\end{itemize}

\noindent 5. Download and extract OpenCV from the following sourceforge location:
\begin{itemize}
	\item[]\href{http://sourceforge.net/projects/opencvlibrary/files/}{$http://sourceforge.net/projects/opencvlibrary/files/$}
\end{itemize}

\noindent 6. Create a build directory within the unzipped file and enter into it:
\begin{itemize}
	\item[]mkdir \textunderscore build
	\item[]cd \textunderscore build
\end{itemize}

\noindent 7. Initiate the build process with the appropriate flags:
\begin{itemize}
	\item[]cmake -D CMAKE\_BUILD\_TYPE=RELEASE -D MAKE\_INSTALL\_PREFIX=/usr/local -D  WITH\_TBB=ON -D BUILD\_NEW\_PYTHON\_SUPPORT=ON -D WITH\_V4L=ON -D INSTALL\_C\_EXAMPLES=ON -D INSTALL\_PYTHON\_EXAMPLES=ON -D BUILD\_EXAMPLES=ON -D WITH\_QT=ON -D WITH\_OPENGL=ON ..
\end{itemize}

\noindent 8.  Compile the built source code:
\begin{itemize}
	\item[]make
\end{itemize}

\noindent 9. Install OpenCV:
\begin{itemize}
	\item[]sudo make install
\end{itemize}
